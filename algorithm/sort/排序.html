    /**
     * 5W数据，4863 左右
     * @param arr
     */
    function bubbleSort(arr) {
        let flag = false
        for (let i = 0; i < arr.length - 1; i++) {
            for (let j = 0; j < arr.length - 1 - i; j++) {
                flag = true
                if (arr[j] > arr[j + 1]) swap(arr,j, j + 1)
            }
            if (!flag) { // 在一趟排序中，依次交换都没发生过
                break
            } else {// 进行过依次交换，重置flag，用于 下次判断
                flag = false
            }
        }
    }

# 选择排序

    /**
     * 找后面所有数中最小的值，放到前面
     * 5W 数据 800ms 左右
     * @param arr
     */
    function selectSort(arr) {
        for (let i = 0; i < arr.length - 1; i++) {
            let min = arr[i]
            let minIndex = i
            for (let j = i+1; j < arr.length; j++) {
                if (min > arr[j]) {
                    minIndex = j
                    min = arr[j]
                }
            }
            if (minIndex !== i) {
                arr[minIndex] = arr[i]
                arr[i] = min
            }
        }
    }

# 插入排序

    /**
     * 5W 数据是 500ms 左右
     * @param arr
     */
    function insertSort(arr) {
        for (let i = 1; i < arr.length; i++) {
            let insertVal = arr[i]
            let insertIndex = i - 1
            while (insertIndex >= 0 && insertVal < arr[insertIndex]) {
                arr[insertIndex + 1] = arr[insertIndex]
                insertIndex--
            }
            // 当退出while循环时，说明插入的位置找到了
            // 这个判断可以不加，执行时间差不多
            if (insertIndex + 1 !== i) {
                arr[insertIndex + 1] = insertVal
            }
        }
    }

# 快排

    /**
     * 二叉树前序遍历
     * 5W 数据 10ms左右 最低 6ms
     * 50W 最低 55
     * @param arr
     */
    function quickSort(arr, low, high) {
        let i = low, j = high
        if (i < j) {
            let temp = arr[i]
            while (i < j) {
                // 先从后往前找第一个比 t 小的
                while (i < j && arr[j] >= temp) j--
                arr[i] = arr[j]
                // 从前往后找第一个比 t 大的
                while (i < j && arr[i] <= temp) i++
                arr[j] = arr[i]
            }
            arr[i] = temp
            quickSort(arr, low, i - 1)
            quickSort(arr, i + 1, high)
        }
    }

# 归并排序

     /**
     * 二叉树后序遍历
     * 5W数据 10ms左右
     * 50W数据 85ms左右
     * @param arr
     */
    let temp = []
    function mergeSort(arr, left, right, temp) {
        if (left < right) {
            let mid = Math.floor((left + right) / 2)
            // 向左递归进行分解
            mergeSort(arr, left, mid, temp)
            // 向右递归进行分解
            mergeSort(arr, mid+1, right, temp)
            // 每分解一次就合并一次
            merge(arr, left, mid, right, temp)
        }
    }
    /**
     * 合并子数组
     * @param arr 待排序的数组
     * @param left 左边有序序列的初始索引
     * @param mid
     * @param right
     * @param temp 中转数组
     */
    function merge(arr, left, mid, right, temp) {
        let i = left //左边有序序列的初始索引
        let j = mid + 1 // 右边有序序列的初始索引
        let t = 0 // 临时数组的 当前 索引
        // 1.先把左右两边的数据按照规则填充到 temp中，直到左右两边的有序序列，有一边处理完毕为止
        while (i <= mid && j <= right) {
            // 左边的有序序列的当前元素，小于等于 右边有序序列的当前元素，然后后移
            if (arr[i] <= arr[j]) {
                temp[t++] = arr[i++]
            } else {
                // 将右边有序序列的当前元素，填充到当前数组，然后 后移
                temp[t++] = arr[j++]
            }
        }
        // 2.把剩余一边的数据全部依次填入temp
        while (i <= mid) {
            temp[t++] = arr[i++]
        }
        while (j <= right) {
            temp[t++] = arr[j++]
        }
        // 3. temp copy to arr
        // 注意：这里不是每次都拷贝所有的
        t = 0
        let tempLeft = left
        /**
         * 假设源数组len = 8
         * 源数组被拆分成若干个小数组，每个数组len=2
         * 第一次合并 tl = 0,right = 1
         * 第二次合并 tl = 2,right = 3
         * 第三次合并 tl = 0,right = 3
         * 最后一次合并才是 tl = 0, right = 7 即将整个temp copy到原数组
         */
        while (tempLeft <= right) {
            arr[tempLeft++] = temp[t++]
        }

    }

# 基数排序

    /**
     * 5W数据，18ms左右
     * 50W数据，80ms左右
     * 500W 600ms左右
     * 桶排序 循环次数是数组中最大数的位数
     * 空间换时间的经典算法，为了防止溢出，每个桶的大小为 要排序的数组长度
     * 准备10个桶（十个数组）
     * 第一轮
     * 1.将每个元素的个位数取出，看下应该放到哪个桶
     * 2.按照桶的顺序(一维数组下标)，依次取出数据，放入原来数组
     *
     * 第二轮
     * 1.将每个元素的十位数取出，看下应该放到哪个桶，如果没有 十位数，补0，即放到下标为0的数组中
     * 2.按照桶的顺序(一维数组下标)，依次取出数据，放入原来数组
     * 。。。
     * 百位、千位===
     * @param arr
     */
    function radixSort(arr) {
        // 准备10个桶
        let bucket = []
        // 记录每个桶中实际存放了多少个数据，用一个一维数组，记录每个桶每次放入的数据个数
        let bucketElementCounts = []
        for (let i = 0; i <= 9; i++) {
            bucket[i] = []
            bucketElementCounts[i] = 0
        }
        // 取数组中最大的值
        let max = arr[0]
        for (let i = 0; i < arr.length; i++) {
            if (arr[i] > max) {
                max = arr[i]
            }
        }
        // 最大的数是几位数
        let maxLength = (max+'').length
        for (let i = 0, n = 1; i < maxLength; i++, n *= 10) {
            for (let j = 0; j < arr.length; j++) {
                // 取出每个元素的个位数
                let digitOfEle = Math.floor(arr[j] / n) % 10
                // 放入到对应的桶中 ,第digitOfEle个桶中第bucketElementCounts[digitOfEle]个位置
                bucket[digitOfEle][bucketElementCounts[digitOfEle]] = arr[j]
                bucketElementCounts[digitOfEle] ++
            }
            // 按照桶的顺序(一维数组下标)，依次取出数据，放入原来数组
            let index = 0
            // 遍历所有的桶，并将桶中的数据，放入原数组
            for (let k = 0; k < bucketElementCounts.length; k++) {
                // 如果桶中有数据，才放入原数组
                if (bucketElementCounts[k] !== 0) {
                    // 第k 个桶中有数据
                    for (let l = 0; l < bucketElementCounts[k]; l++) {
                        // 第k个桶的第l个元素
                        arr[index++] = bucket[k][l]
                    }
                }
                // 每轮处理后，需要将bucketElementCounts[k] = 0
                bucketElementCounts[k] = 0
            }
        }
    }

# 堆排序

i 从 0 开始
左子树 arr[2*i+1]  右子树 arr[2*i+2]
父节点下标 (i-1) / 2
大顶堆：
    特点 arr[i] >= arr[2*i+1] && arr[i] >= arr[2*i+2]
小顶堆： 
    特点 arr[i] <= arr[2*i+1] && arr[i] <= arr[2*i+2]

i 从 1 开始 数组第一个位置留空，方便计算
左子树 arr[i*2]   右子树 arr[i*2+1]
父节点下标 i/2

将待排序序列构造成一个大顶堆，此时整个序列的最大值就是堆顶的根节点。将其与末位元素交换， 此时末尾元素就是最大值。然后将剩余n-1个元素重新构造成一个堆，这样会得到n个元素的次小值

/**
     * 50W数据 70ms 左右
     * 500W数据 1100 ms 左右
     * @param arr
     */
    function heapSort(arr) {
        for (let i = parseInt(arr.length / 2 - 1); i >= 0; i--) {
            heapify(arr, i, arr.length)
        }
        let temp = 0
        for (let j = arr.length -1; j > 0; j--) {
            temp = arr[j]
            arr[j] = arr[0]
            arr[0] = temp
            heapify(arr, 0, j)
        }
    }
    /** 构建大顶堆
     * @param arr 待调整的数组
     * @param i 非叶子节点在数组中的索引
     * @param len 堆多少个元素进行调整， len在逐渐减少
     */
    function heapify(arr, i, len) {
        let temp = arr[i]
        // k 指向 i 节点的左子节点
        for (let k = 2*i+1; k < len; k = k*2+1) {
            if (k + 1 < len && arr[k] < arr[k + 1]) { // 说明 以 i 为根的 左子节点 小于 右子节点
                k++ // k 指向 右子节点
            }
            if (arr[k] > temp) { // 子节点中最大的值 大于 父节点
                arr[i] = arr[k] // 把较大的值 赋值给当前节点
                i = k // i 指向 k 继续比较
            } else {
                break
            }
        }
        // 到这里，说明 以 i 为 根的大顶堆已经构成
        arr[i] = temp
    }

// 下标从1 开始，速度差不多
function heapify2(arr,i,len) {
    let left = 2 * i + 1
    while (left < len) {
        // 找以i为根的子节点最大值下标
        let largest = left + 1 < len && arr[left + 1] > arr[left] ? left + 1 : left
        // 子节点中的最大值与当前值比较
        largest = arr[i] > arr[largest] ? i : largest
        if (largest === i) {
            // 当前节点已经是最大值
            break
        }
        swap(arr, i, largest)
        i = largest
        left = 2 * i + 1
    }
}
function heapSort2(arr) {
    let beginIndex = (arr.length - 2) >> 1
    for (let i = beginIndex; i >= 0; i--) {
        heapify2(arr, i,arr.length - 1)
    }
    for (let i = arr.length - 1; i > 0; i--) {
        swap(arr,0, i-1)
        heapify2(arr, 0, i-1)
    }

}
