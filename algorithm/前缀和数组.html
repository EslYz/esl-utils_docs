    # 二维矩阵中的前缀和

    class NumMatrix {
    constructor(matrix) {
        let m = matrix.length, n = matrix[0].length
        this.preSum = []
        for (let i = 0; i < m; i++) {
            // 遍历行，并初始化preSum
            this.preSum[i] = []
            for (let j = 0; j < n; j++) {
                // 处理列
                this.preSum[i][j] = matrix[i][j]
                if (i > 0) {
                    this.preSum[i][j] += this.preSum[i-1][j]
                }
                if (j > 0) {
                    this.preSum[i][j] += this.preSum[i][j-1]
                }
                if (i > 0 && j > 0) {
                    // 去掉重复加上的值
                    this.preSum[i][j] -= this.preSum[i-1][j-1]
                }
            }
        }
    }
    sumRegion(row1, col1, row2, col2) {
        let temp = this.preSum[row2][col2]
        if (row1 > 0) {
            temp -= this.preSum[row1-1][col2]
        }
        if (col1 > 0) {
            temp -= this.preSum[row2][col1-1]
        }
        if (row1 > 0 && col1 > 0) {
            temp += this.preSum[row1-1][col1-1]
        }
        return temp
    }
}

    # 和为k的数组

    给你一个整数数组 nums 和一个整数 k ，请你统计并返回该数组中和为 k 的连续子数组的个数。
示例 1：
输入：nums = [1,1,1], k = 2
输出：2
示例 2：
输入：nums = [1,2,3], k = 3
输出：2

    function subarraySum(nums, k) {
    let res = 0
    let map = new Map()
    map.set(0, 1)
    let sum = 0
    for (let i = 0; i < nums.length; i++) {
        sum += nums[i]
        let diff = sum-k
        if (map.has(diff)) {
            res += map.get(diff)
        }
        let count = map.get(sum)
        if(!count) count=0
        map.set(sum, count+1)
    }
    return res;
}
