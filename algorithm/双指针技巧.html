# 快慢指针的常见算法


快慢指针一般都初始化指向链表的头节点 head，前进时快指针 fast在前，慢指针 slow在后，巧妙的解决一些链表中的问题。

滑动窗口 这也许是快慢指针的最高境界了
  1. 判断链表中是否包含环
function hasCycle(head) {
    let slow = fast = head
    while (fast !== null && fast.next !== null) {
        fast = fast.next.next
        slow = slow.next
        if (slow === fast) return true
    }
    return false
}

  1. 已知链表中含有环，返回这个环的起始位置
function detectCycle(head) {
    let slow = fast = head
    while (fast !== null && fast.next !== null) {
        slow = slow.next
        fast = fast.next.next
        if (slow === fast) break
    }
    if (fast === null || fast.next === null) return null
    slow = head
    while (slow !== head) {
        slow = slow.next
        fast = fast.next
    }
    return slow
}

function middleNode(head) {
    let slow = fast = head
    while (fast !== null && fast.next !== null) {
        slow = slow.next
        fast = fast.next.next
    }
    return slow
}

  1. 删除链表倒数第 n 个节点，n小于 链表长度
function removeNthFromEnd(head, n) {
    let dummyNode = new ListNode(-1, head)
    let slow = fast = head
    while (n-- > 0) {
        fast = fast.next
    }
    // fast = null 说明倒数第n个节点就是头节点
    if (fast === null) {
        return head.next
    }
    // let pre = slow
    // while(fast !== null) {
    //     fast = fast.next
    //     pre = slow
    //     slow = slow.next
    // }
    // pre.next = slow.next
    // 减少一次遍历，少用一个临时变量
    while (fast.next !== null){
        fast = fast.next
        slow = slow.next
    }
    // slow.next 就是要删除的倒数第 n个节点
    slow.next = slow.next.next
    return dummyNode.next
}
